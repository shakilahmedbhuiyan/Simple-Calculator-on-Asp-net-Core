public JsonResult Calculat(string firstNumber, string secondNumber, string action)
        {
            CalculatorActionModel model = new CalculatorActionModel();

            double a = Convert.ToDouble(firstNumber);
            double b = Convert.ToDouble(secondNumber);
            double c = 0;
            switch (action)
            {
                case "add":
                    c = a + b;
                    break;
                case "sub":
                    c = a - b;
                    break;
                case "mul":
                    c = a * b;
                    break;
                case "div":
                    c = a / b;
                    break;
            }
            model.firstNumber = a;
            model.secondNumber = b;
            model.result = c;

            ViewBag.Result = c;
            return Json(model);
        }
